Installation

# Install with npm
npm install -g @sammons/code-outline-cli

# Or use with npx
npx @sammons/code-outline-cli "src/**/*.ts"

# For monorepo development
pnpm install
pnpm build

Usage

# Analyze TypeScript files
code-outline "src/**/*.ts"

# Export as JSON
code-outline "*.js" --format json

# Limit depth and export as YAML
code-outline "src/**/*.tsx" --depth 2 --format yaml

# Show all nodes (not just named ones)
code-outline "index.js" --all
